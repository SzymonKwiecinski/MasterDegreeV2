\textbf{Knapsack Problem Formulation:}

\begin{align*}
    \text{Maximize} \quad & \sum_{k=1}^{K} \text{value}_k \cdot x_k \\
    \text{Subject to} \quad & \sum_{k=1}^{K} \text{size}_k \cdot x_k \leq C \\
    & x_k \in \{0, 1\} \quad \text{for } k = 1, \ldots, K
\end{align*}
