\section*{Linear Programming Model}

\textbf{Objective:}
\[
\min \, d
\]

\textbf{Subject to:}
\[
y_k - (bx_k + a) \leq d, \quad k = 1, 2, \ldots, K
\]
\[
(bx_k + a) - y_k \leq d, \quad k = 1, 2, \ldots, K
\]

\textbf{Variables:}
\[
a \, (\text{intercept}), \quad b \, (\text{slope}), \quad d \, (\text{maximum deviation})
\]
