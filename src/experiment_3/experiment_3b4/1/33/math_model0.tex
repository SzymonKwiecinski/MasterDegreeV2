\textbf{Mixed Integer Programming Formulation for the Knapsack Problem}

\textbf{Objective:}
\[
\text{Maximize} \quad \sum_{k=1}^{K} \text{value}_k \cdot x_k
\]

\textbf{Subject to:}
\[
\sum_{k=1}^{K} \text{size}_k \cdot x_k \leq C
\]

\[
x_k \in \{0, 1\}, \quad \forall k \in K
\]
