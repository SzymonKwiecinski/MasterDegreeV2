\textbf{Linear Programming Model for Finding the Chebychev Center}

\[
\begin{align*}
\text{Maximize} \quad & r \\
\text{subject to} \quad & a_i^T y + \|a_i\| r \leq b_i, \quad i = 1, \ldots, m, \\
& y \in \mathbb{R}^N, \\
& r \in \mathbb{R}.
\end{align*}
\]
