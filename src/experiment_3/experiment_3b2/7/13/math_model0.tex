\textbf{Linear Programming Problem:}

\[
\begin{align*}
\text{maximize} \quad & r \\
\text{subject to} \quad & \mathbf{a_i}^T \mathbf{y} + r \|\mathbf{a_i}\|_2 \leq b_i, \quad \forall i = 1, \ldots, m \\
& \mathbf{y} \in \mathbb{R}^N, \\
& r \in \mathbb{R}.
\end{align*}
\]
