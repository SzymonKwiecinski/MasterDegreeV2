import pulp

#### Data
data = {'NumObs': 19, 'Y': [1.0, 0.9, 0.7, 1.5, 2.0, 2.4, 3.2, 2.0, 2.7, 3.5, 1.0, 4.0, 3.6, 2.7, 5.7, 4.6, 6.0, 6.8, 7.3], 'X': [0.0, 0.5, 1.0, 1.5, 1.9, 2.5, 3.0, 3.5, 4.0, 4.5, 5.0, 5.5, 6.0, 6.6, 7.0, 7.6, 8.5, 9.0, 10.0]}
y_values = data['Y']
x_values = data['X']
num_obs = data['NumObs']

#### Problem
problem = pulp.LpProblem("Minimize_Maximum_Deviation", pulp.LpMinimize)

#### Variables
a = pulp.LpVariable('intercept', lowBound=None, upBound=None, cat='Continuous')
b = pulp.LpVariable('slope', lowBound=None, upBound=None, cat='Continuous')
d = pulp.LpVariable('max_deviation', lowBound=0, cat='Continuous')

#### Objective
problem += d, "Minimize_Maximum_Deviation"

#### Constraints
for k in range(num_obs):
    #### y_k - (a + b*x_k) <= d
    problem += y_values[k] - (a + b * x_values[k]) <= d
    
    #### -(y_k - (a + b*x_k)) <= d
    problem += -(y_values[k] - (a + b * x_values[k])) <= d

#### Solve
problem.solve()

#### Output
solution = {"intercept": a.varValue, "slope": b.varValue}
print(solution)
print(f' (Objective Value): <OBJ>{pulp.value(problem.objective)}</OBJ>')
